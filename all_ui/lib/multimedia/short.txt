// Video model
class VideoItem {
  final String id;
  final String title;
  final String channelName;
  final String thumbnailUrl;
  final Duration duration;
  final int likes;
  final int views;

  VideoItem({
    required this.id,
    required this.title,
    required this.channelName,
    required this.thumbnailUrl,
    required this.duration,
    required this.likes,
    required this.views,
  });
}

// Video state model
class VideoPlayerState {
  final List<VideoItem> videos;
  final int currentIndex;
  final bool isPlaying;
  final bool isMuted;
  final bool isLoading;

  VideoPlayerState({
    required this.videos,
    required this.currentIndex,
    required this.isPlaying,
    required this.isMuted,
    required this.isLoading,
  });

  VideoPlayerState copyWith({
    List<VideoItem>? videos,
    int? currentIndex,
    bool? isPlaying,
    bool? isMuted,
    bool? isLoading,
  }) {
    return VideoPlayerState(
      videos: videos ?? this.videos,
      currentIndex: currentIndex ?? this.currentIndex,
      isPlaying: isPlaying ?? this.isPlaying,
      isMuted: isMuted ?? this.isMuted,
      isLoading: isLoading ?? this.isLoading,
    );
  }
}

// Video player state notifier
class VideoPlayerNotifier extends StateNotifier<VideoPlayerState> {
  late YoutubePlayerController _controller;

  VideoPlayerNotifier()
    : super(
        VideoPlayerState(
          videos: sampleVideos,
          currentIndex: 0,
          isPlaying: false,
          isMuted: false,
          isLoading: true,
        ),
      ) {
    _initializePlayer();
  }

  YoutubePlayerController get controller => _controller;

  void _initializePlayer() {
    if (state.videos.isNotEmpty) {
      _controller = YoutubePlayerController(
        initialVideoId: state.videos[state.currentIndex].id,
        flags: const YoutubePlayerFlags(
          autoPlay: true,
          mute: false,
          loop: false,
          hideControls: true,
          controlsVisibleAtStart: false,
        ),
      );

      _controller.addListener(() {
        if (_controller.value.isReady && state.isLoading) {
          state = state.copyWith(isLoading: false, isPlaying: true);
        }

        if (_controller.value.playerState == PlayerState.ended) {
          nextVideo();
        }
      });
    }
  }

  void setCurrentIndex(int index) {
    if (index != state.currentIndex &&
        index >= 0 &&
        index < state.videos.length) {
      state = state.copyWith(currentIndex: index, isLoading: true);
      _controller.load(state.videos[index].id);
    }
  }

  void nextVideo() {
    if (state.currentIndex < state.videos.length - 1) {
      final newIndex = state.currentIndex + 1;
      state = state.copyWith(currentIndex: newIndex, isLoading: true);
      _controller.load(state.videos[newIndex].id);
    }
  }

  void previousVideo() {
    if (state.currentIndex > 0) {
      final newIndex = state.currentIndex - 1;
      state = state.copyWith(currentIndex: newIndex, isLoading: true);
      _controller.load(state.videos[newIndex].id);
    }
  }

  void togglePlayPause() {
    if (state.isPlaying) {
      _controller.pause();
      state = state.copyWith(isPlaying: false);
    } else {
      _controller.play();
      state = state.copyWith(isPlaying: true);
    }
  }

  void toggleMute() {
    if (state.isMuted) {
      _controller.unMute();
      state = state.copyWith(isMuted: false);
    } else {
      _controller.mute();
      state = state.copyWith(isMuted: true);
    }
  }

  @override
  void dispose() {
    _controller.dispose();
    super.dispose();
  }
}

// Provider
final videoPlayerProvider =
    StateNotifierProvider<VideoPlayerNotifier, VideoPlayerState>((ref) {
      return VideoPlayerNotifier();
    });

// Main screen
class ContinuousYouTubePlayerScreen extends ConsumerStatefulWidget {
  const ContinuousYouTubePlayerScreen({Key? key}) : super(key: key);

  @override
  ConsumerState<ContinuousYouTubePlayerScreen> createState() =>
      _ContinuousYouTubePlayerScreenState();
}

class _ContinuousYouTubePlayerScreenState
    extends ConsumerState<ContinuousYouTubePlayerScreen> {
  late PageController _pageController;
  bool _showControls = false;

  @override
  void initState() {
    super.initState();
    _pageController = PageController();
  }

  @override
  void dispose() {
    _pageController.dispose();
    super.dispose();
  }

  void _toggleControls() {
    setState(() {
      _showControls = !_showControls;
    });
  }

  @override
  Widget build(BuildContext context) {
    final videoState = ref.watch(videoPlayerProvider);
    final videoNotifier = ref.read(videoPlayerProvider.notifier);

    return Scaffold(
      backgroundColor: Colors.black,
      body: SafeArea(
        child: PageView.builder(
          controller: _pageController,
          scrollDirection: Axis.vertical,
          itemCount: videoState.videos.length,
          onPageChanged: (index) {
            // Update the video state to match the page index
            videoNotifier.setCurrentIndex(index);
          },
          itemBuilder: (context, index) {
            return VideoPlayerWidget(
              video: videoState.videos[index],
              isCurrentVideo: index == videoState.currentIndex,
              showControls: _showControls,
              onTap: _toggleControls,
            );
          },
        ),
      ),
    );
  }
}

// Video player widget
class VideoPlayerWidget extends ConsumerWidget {
  final VideoItem video;
  final bool isCurrentVideo;
  final bool showControls;
  final VoidCallback onTap;

  const VideoPlayerWidget({
    Key? key,
    required this.video,
    required this.isCurrentVideo,
    required this.showControls,
    required this.onTap,
  }) : super(key: key);

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final videoState = ref.watch(videoPlayerProvider);
    final videoNotifier = ref.read(videoPlayerProvider.notifier);

    return GestureDetector(
      onTap: onTap,
      child: Container(
        width: double.infinity,
        height: double.infinity,
        color: Colors.black,
        child: Stack(
          children: [
            // Video player
            if (isCurrentVideo)
              Center(
                child: AspectRatio(
                  aspectRatio: 9 / 16,
                  child: YoutubePlayer(
                    controller: videoNotifier.controller,
                    showVideoProgressIndicator: false,
                    bottomActions: [],
                    topActions: [],
                  ),
                ),
              )
            else
              // Thumbnail for non-current videos
              Container(
                width: double.infinity,
                height: double.infinity,
                decoration: BoxDecoration(
                  image: DecorationImage(
                    image: NetworkImage(video.thumbnailUrl),
                    fit: BoxFit.cover,
                  ),
                ),
              ),

            // Loading indicator
            if (videoState.isLoading && isCurrentVideo)
              const Center(
                child: CircularProgressIndicator(color: Colors.white),
              ),

            // Video controls overlay
            if (showControls && isCurrentVideo)
              VideoControlsOverlay(
                video: video,
                isPlaying: videoState.isPlaying,
                isMuted: videoState.isMuted,
                onPlayPause: videoNotifier.togglePlayPause,
                onMute: videoNotifier.toggleMute,
              ),

            // Video info overlay
            VideoInfoOverlay(video: video),

            // Side action buttons
            SideActionButtons(video: video),
          ],
        ),
      ),
    );
  }
}

// Video controls overlay
class VideoControlsOverlay extends StatelessWidget {
  final VideoItem video;
  final bool isPlaying;
  final bool isMuted;
  final VoidCallback onPlayPause;
  final VoidCallback onMute;

  const VideoControlsOverlay({
    Key? key,
    required this.video,
    required this.isPlaying,
    required this.isMuted,
    required this.onPlayPause,
    required this.onMute,
  }) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return Container(
      color: Colors.black.withOpacity(0.3),
      child: Center(
        child: Row(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            IconButton(
              onPressed: onPlayPause,
              icon: Icon(
                isPlaying ? Icons.pause : Icons.play_arrow,
                color: Colors.white,
                size: 64,
              ),
            ),
            const SizedBox(width: 32),
            IconButton(
              onPressed: onMute,
              icon: Icon(
                isMuted ? Icons.volume_off : Icons.volume_up,
                color: Colors.white,
                size: 32,
              ),
            ),
          ],
        ),
      ),
    );
  }
}

// Video info overlay
class VideoInfoOverlay extends StatelessWidget {
  final VideoItem video;

  const VideoInfoOverlay({Key? key, required this.video}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return Positioned(
      bottom: 80,
      left: 16,
      right: 80,
      child: Column(
        crossAxisAlignment: CrossAxisAlignment.start,
        children: [
          Text(
            video.title,
            style: const TextStyle(
              color: Colors.white,
              fontSize: 16,
              fontWeight: FontWeight.bold,
            ),
            maxLines: 2,
            overflow: TextOverflow.ellipsis,
          ),
          const SizedBox(height: 8),
          Text(
            video.channelName,
            style: const TextStyle(color: Colors.white70, fontSize: 14),
          ),
          const SizedBox(height: 4),
          Text(
            '${_formatNumber(video.views)} views',
            style: const TextStyle(color: Colors.white70, fontSize: 12),
          ),
        ],
      ),
    );
  }

  String _formatNumber(int number) {
    if (number >= 1000000) {
      return '${(number / 1000000).toStringAsFixed(1)}M';
    } else if (number >= 1000) {
      return '${(number / 1000).toStringAsFixed(1)}K';
    }
    return number.toString();
  }
}

// Side action buttons
class SideActionButtons extends StatelessWidget {
  final VideoItem video;

  const SideActionButtons({Key? key, required this.video}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return Positioned(
      right: 16,
      bottom: 80,
      child: Column(
        children: [
          _buildActionButton(
            icon: Icons.thumb_up_outlined,
            label: _formatNumber(video.likes),
            onTap: () {},
          ),
          const SizedBox(height: 24),
          _buildActionButton(
            icon: Icons.comment_outlined,
            label: 'Comment',
            onTap: () {},
          ),
          const SizedBox(height: 24),
          _buildActionButton(
            icon: Icons.share_outlined,
            label: 'Share',
            onTap: () {},
          ),
          const SizedBox(height: 24),
          _buildActionButton(
            icon: Icons.bookmark_outline,
            label: 'Save',
            onTap: () {},
          ),
        ],
      ),
    );
  }

  Widget _buildActionButton({
    required IconData icon,
    required String label,
    required VoidCallback onTap,
  }) {
    return GestureDetector(
      onTap: onTap,
      child: Column(
        children: [
          Container(
            padding: const EdgeInsets.all(12),
            decoration: BoxDecoration(
              color: Colors.black.withOpacity(0.3),
              shape: BoxShape.circle,
            ),
            child: Icon(icon, color: Colors.white, size: 24),
          ),
          const SizedBox(height: 4),
          Text(
            label,
            style: const TextStyle(color: Colors.white, fontSize: 10),
          ),
        ],
      ),
    );
  }

  String _formatNumber(int number) {
    if (number >= 1000000) {
      return '${(number / 1000000).toStringAsFixed(1)}M';
    } else if (number >= 1000) {
      return '${(number / 1000).toStringAsFixed(1)}K';
    }
    return number.toString();
  }
}
